    class GeoEntity {
	    +string id
	    +string name
	    +Geometry geometry
	    +string[] sourceRefs

    class CoalUnit {
	    +string unitCode
	    +string unitName
	    +string coalRank
	    +string formation
    }

    class Colliery {
	    +string operator
	    +string status
	    +number productionTons
	    +number openedYear
	    +number closedYear
    }

    class MonitoringStation {
	    +string stationCode
	    +string agency
	    +string stationType
    }

    class WaterQualitySample {
	    +string sampleId
	    +string stationId
	    +string timestamp
	    +string parameter
	    +number value
	    +string unit
    }

    class StreamSegment {
	    +string reachId
	    +string watershedId
	    +number lengthKm
	    +string impairedStatus
	    +string impairmentCause
    }

    class AMDHarm {
	    +string harmId
	    +string label
	    +string timeWindowStart
	    +string timeWindowEnd
	    +string severity
	    +string[] sourceCollieryIds
	    +string[] supportingStationIds
	    +string[] affectedReachIds
	    +object keyMetrics
	    +string notes
	    +summarize(samples)
	    +toEvidencePacket()
    }
